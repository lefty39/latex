\section{Topological spaces}
\begin{description}
    \item[topology] A set $T$ of subsets of $X$ such that \begin{enumerate}
        \item $X \in T$
        \item $\varnothing \in T$
        \item $T$ is closed under \emph{arbitrary unions} and \emph{finite intersections}
    \end{enumerate}
    \item[basis] A collection $\mathcal{B}$ of subsets of $X$ such that \begin{enumerate}
        \item $\forall x \in X,\exists B\in \mathcal{B}: x\in B$
        \item $\forall B_{1},B_{2} \in \mathcal{B},x \in B_{1}\cap B_{2}\implies \exists B_{3}\in\mathcal{B}:x \in B_3\subseteq B_{1}\cap B_{2}$
    \end{enumerate}
    The set $T_\mathcal{B}=\left\{ \bigcup \mathcal{F} \mid \mathcal{F} \subset \mathcal{B} \right\}$ is the topology generated by $\mathcal{B}$.

    \item[fineness] $T\text{ is finer than }S \iff S \subset T$
    \item[subspace] $(Y,T_Y)$ where $Y\subset X$ and $T_Y = \left\{ Y \cap U \mid U \in T\right\}$
    \item[closure] $\overline{A} = \mathrm{Cl}(A) = \bigcap \left\{ U\mid \complement_{X}U\in T\land A\subset U \right\}$ is the smallest closed set containing $A$
    \item[interior] $\mathrm{Int}(A)=\bigcup \left\{ U\mid U\in T \land U\subset A \right\}$ is the largest open set contained in $A$
\end{description}

\section{Maps}
\begin{description}
    \item[continuous] a map $f:X \to Y$ between topological spaces $(X,T)$ and $(Y,S)$ if $\forall V \in S, f^{-1}(V)\in T$
    \item[homeomorphism] a bijection $f:X\to Y$ if both $f$ and $f^{-1}$ are continuous
\end{description}
\section{Metric spaces}
\begin{description}
    \item[metric] on a set $X$, the function $d:X \times X \to \mathbb{R}$ such that, $\forall x,y,z\in X,$ \begin{enumerate}
        \item $d(x,y) \geq 0$
        \item $d(x,y) = 0 \iff x = y$
        \item $d(x,y)=d(y,x)$
        \item $d(x,z) \leq d(x,y)+d(y,z)$
    \end{enumerate}
    \item A metric \emph{induces} the topology generated by the basis of all open balls using $d$.
    \item[metrizable] a topological space if there is a metric that induces its topology

\end{description}
