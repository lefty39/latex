\section{Topological Spaces}
\begin{description}
    \item[topology] A set $T$ of subsets of $X$ such that \begin{enumerate}
        \item $X \in T$
        \item $\varnothing \in T$
        \item $T$ is closed under \emph{arbitrary unions} and \emph{finite intersections}
    \end{enumerate}
    \item[basis] A collection $\mathcal{B}$ of subsets of $X$ such that \begin{enumerate}
        \item $\forall x \in X,\exists B\in \mathcal{B}: x\in B$
        \item $\forall B_{1},B_{2} \in \mathcal{B},x \in B_{1}\cap B_{2}\implies \exists B_{3}\in\mathcal{B}:x \in B_3\subseteq B_{1}\cap B_{2}$
    \end{enumerate}
    The set $T_\mathcal{B}=\left\{ \bigcup \mathcal{F} \mid \mathcal{F} \subset \mathcal{B} \right\}$

    \item[fineness] $T\text{ is finer than }S \iff S \subset T$
    \item[subspace] $(Y,T_Y)$ where $Y\subset X$ and $T_Y = \left\{ Y \cap U \mid U \in T\right\}$
    \item[closure] $\overline{A} = \mathrm{Cl}(A) = \bigcap \left\{ U\mid \complement_{X}U\in T\land A\subset U \right\}$ is the smallest closed set containing $A$
    \item[interior] $\mathrm{Int}(A)=\bigcup \left\{ U\mid U\in T \land U\subset A \right\}$ is the largest open set contained in $A$
\end{description}
